Examples

Creating a window

        
from margray_2d import *


margray = Margray()
margray.create_screen()

while True:
    margray.fps().setMaxFPS(NORMAL_FPS)
    events = EventHandler(margray.run())
    events.enableAutoQuit(margray)
    margray.update()

Creating a sprite

        
from margray_2d import *

margray = Margray()
margray.create_screen()
sprite = margray.sprite(color=(0,155,120),position=(10,10),size=(50,50))
while True:
    margray.fps().setMaxFPS(NORMAL_FPS)
    events = EventHandler(margray.run())
    events.enableAutoQuit(margray)
    sprite.draw()
    margray.update()

Creating an animated sprite

        
from margray_2d import *

margray = Margray()
margray.create_screen()
Images = ["","","",""]
animated_sprite = margray.animated_sprite(image_paths=Images,x=50,y=50,size=None,startingIndex=0)
while True:
    margray.fps().setMaxFPS(NORMAL_FPS)
    events = EventHandler(margray.run())
    events.enableAutoQuit(margray)
    animated_sprite.draw()
    margray.update()
