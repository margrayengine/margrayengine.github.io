Creating a window

        
from margray_2d import *


margray = Margray()
margray.create_screen()

while True:
    margray.fps().setMaxFPS(NORMAL_FPS)
    events = EventHandler(margray.run())
    events.enableAutoQuit(margray)
    margray.update()

Creating a sprite

        
from margray_2d import *

margray = Margray()
margray.create_screen()
sprite = margray.sprite(color=(0,155,120),position=(10,10),size=(50,50))
while True:
    margray.fps().setMaxFPS(NORMAL_FPS)
    events = EventHandler(margray.run())
    events.enableAutoQuit(margray)
    sprite.draw()
    margray.update()

Creating an animated sprite

Working with Keys

        
from margray_2d import *

margray = Margray()
margray.create_screen()

keyTime = Time()

while True:
    margray.fps().setMaxFPS(NORMAL_FPS)
    events = EventHandler(margray.run())
    events.enableAutoQuit(margray)
    keyW = events.isKey(Keys.KEY_W)
    #Detecting if the key is pressed and not held
    if keyW.pressed():
        print("Key 'W' is pressed")
    #Detecting if the key is released
    if keyW.released():
        print("Key 'W' is released") 
    #Detecting if the key is pressed and held       
    if keyW.held():
        print("Key 'W' is held") 
    #Detecting if the key is held for 1 second    
    if keyW.held_for(keyTime,1000):
        print("Key 'W' is held for 1 second")        
    margray.update()
